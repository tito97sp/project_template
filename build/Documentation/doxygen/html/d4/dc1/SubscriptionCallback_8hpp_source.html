70  bool registerCallback()
71  {
72  if (_subscription.get_node() && _subscription.get_node()->register_callback(this)) {
73  // registered
74  _registered = true;
75 
76  } else {
77  // force topic creation by subscribing with old API
78  int fd = orb_subscribe_multi(_subscription.get_topic(), _subscription.get_instance());
79 
80  // try to register callback again
81  if (_subscription.subscribe()) {
82  if (_subscription.get_node() && _subscription.get_node()->register_callback(this)) {
83  _registered = true;
84  }
85  }
86 
87  orb_unsubscribe(fd);
88  }
89 
90  return _registered;
91  }
92 
93  void unregisterCallback()
94  {
95  if (_subscription.get_node()) {
96  _subscription.get_node()->unregister_callback(this);
97  }
98 
99  _registered = false;
100  }

106  bool ChangeInstance(uint8_t instance)
107  {
108  bool ret = false;
109 
110  if (instance != get_instance()) {
111  const bool registered = _registered;
112 
113  if (registered) {
114  unregisterCallback();
115  }
116 
117  if (_subscription.ChangeInstance(instance)) {
118  ret = true;
119  }
120 
121  if (registered) {
122  registerCallback();
123  }
124 
125  } else {
126  // already on desired index
127  return true;
128  }
