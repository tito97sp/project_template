75 int uORB::DeviceNode::open(file *filp)
76 {
77  /* is this a publisher? */
78  if (filp->f_oflags == O_WRONLY) {
79 
80  lock();
81  mark_as_advertised();
82  unlock();
83 
84  /* now complete the open */
85  return CDev::open(filp);
86  }
87 
88  /* is this a new subscriber? */
89  if (filp->f_oflags == O_RDONLY) {
90 
91  /* allocate subscriber data */
92  SubscriptionInterval *sd = new SubscriptionInterval(_meta, 0, _instance);
93 
94  if (nullptr == sd) {
95  return -ENOMEM;
96  }
97 
98  filp->f_priv = (void *)sd;
99 
100  int ret = CDev::open(filp);
101 
102  if (ret != 0) {
103  //PX4_ERR("CDev::open failed");
104  delete sd;
105  }
106 
107  return ret;
108  }
109 
110  if (filp->f_oflags == 0) {
111  return CDev::open(filp);
112  }
113 
114  /* can only be pub or sub, not both */
115  return -EINVAL;
116 }

128 bool uORB::DeviceNode::copy(void *dst, unsigned &generation)
129 {
130  if ((dst != nullptr) && (_data != nullptr)) {
131  if (_queue_size == 1) {
132  ATOMIC_ENTER;
133  memcpy(dst, _data, _meta->o_size);
134  generation = _generation.load();
135  ATOMIC_LEAVE;
136  return true;
137 
138  } else {
139  ATOMIC_ENTER;
140  const unsigned current_generation = _generation.load();
141 
142  if (current_generation > generation + _queue_size) {
143  // Reader is too far behind: some messages are lost
144  generation = current_generation - _queue_size;
145  }
146 
147  if ((current_generation == generation) && (generation > 0)) {
148  /* The subscriber already read the latest message, but nothing new was published yet.
149  * Return the previous message
150  */
151  --generation;
152  }
153 
154  memcpy(dst, _data + (_meta->o_size * (generation % _queue_size)), _meta->o_size);
155  ATOMIC_LEAVE;
156 
157  if (generation < current_generation) {
158  ++generation;
159  }
160 
161  return true;
162  }
163  }
164 
165  return false;
166 }

178 ssize_t uORB::DeviceNode::write(file *filp, const char *buffer, size_t buflen)
179 {
180  /*
181  * Writes are legal from interrupt context as long as the
182  * object has already been initialised from thread context.
183  *
184  * Writes outside interrupt context will allocate the object
185  * if it has not yet been allocated.
186  *
187  * Note that filp will usually be NULL.
188  */
189  if (nullptr == _data) {
190 
191  if (!up_interrupt_context()) {
192 
193  lock();
194 
195  /* re-check size */
196  if (nullptr == _data) {
197  _data = new uint8_t[_meta->o_size * _queue_size];
198  }
199 
200  unlock();
201  }
202 
203  /* failed or could not allocate */
204  if (nullptr == _data) {
205  return -ENOMEM;
206  }
207  }
208 
209  /* If write size does not match, that is an error */
210  if (_meta->o_size != buflen) {
211  return -EIO;
212  }
213 
214  /* Perform an atomic copy. */
215  ATOMIC_ENTER;
216  /* wrap-around happens after ~49 days, assuming a publisher rate of 1 kHz */
217  unsigned generation = _generation.fetch_add(1);
218 
219  memcpy(_data + (_meta->o_size * (generation % _queue_size)), buffer, _meta->o_size);
220 
221  // callbacks
222  for (auto item : _callbacks) {
223  item->call();
224  }
225 
226  ATOMIC_LEAVE;
227 
228  /* notify any poll waiters */
229  poll_notify(POLLIN);
230 
231  return _meta->o_size;
232 }

234 int uORB::DeviceNode::ioctl(file *filp, int cmd, unsigned long arg)
235 {
236  switch (cmd) {
237  case ORBIOCUPDATED: {
238  ATOMIC_ENTER;
239  *(bool *)arg = filp_to_subscription(filp)->updated();
240  ATOMIC_LEAVE;
241  return 0;
242  }
243 
244  case ORBIOCSETINTERVAL:
245  filp_to_subscription(filp)->set_interval_us(arg);
246  return 0;
247 
248  case ORBIOCGADVERTISER:
249  *(uintptr_t *)arg = (uintptr_t)this;
250  return 0;
251 
252  case ORBIOCSETQUEUESIZE: {
253  lock();
254  int ret = update_queue_size(arg);
255  unlock();
256  return ret;
257  }
258 
259  case ORBIOCGETINTERVAL:
260  *(unsigned *)arg = filp_to_subscription(filp)->get_interval_us();
261  return 0;
262 
263  case ORBIOCISADVERTISED:
264  *(unsigned long *)arg = _advertised;
265 
266  return 0;
267 
268  default:
269  /* give it to the superclass */
270  return CDev::ioctl(filp, cmd, arg);
271  }

274 ssize_t uORB::DeviceNode::publish(const orb_metadata *meta, orb_advert_t handle, const void *data)
275 {
276  uORB::DeviceNode *devnode = (uORB::DeviceNode *)handle;
277  int ret;
278 
279  /* check if the device handle is initialized and data is valid */
280  if ((devnode == nullptr) || (meta == nullptr) || (data == nullptr)) {
281  errno = EFAULT;
282  return -1;
283  }
284 
285  /* check if the orb meta data matches the publication */
286  if (devnode->_meta != meta) {
287  errno = EINVAL;
288  return -1;
289  }
290 
291  /* call the devnode write method with no file pointer */
292  ret = devnode->write(nullptr, (const char *)data, meta->o_size);
293 
294  if (ret < 0) {
295  errno = -ret;
296  return -1;
297  }
298 
299  if (ret != (int)meta->o_size) {
300  errno = EIO;
301  return -1;
302  }
303 
304 #ifdef ORB_COMMUNICATOR
305  /*
306  * if the write is successful, send the data over the Multi-ORB link
307  */
308  uORBCommunicator::IChannel *ch = uORB::Manager::get_instance()->get_uorb_communicator();
309 
310  if (ch != nullptr) {
311  if (ch->send_message(meta->o_name, meta->o_size, (uint8_t *)data) != 0) {
312  PX4_ERR("Error Sending [%s] topic data over comm_channel", meta->o_name);
313  return -1;
314  }
315  }
316 
317 #endif /* ORB_COMMUNICATOR */
318 
319  return 0;
320 }

322 int uORB::DeviceNode::unadvertise(orb_advert_t handle)
323 {
324  if (handle == nullptr) {
325  return -EINVAL;
326  }
327 
328  uORB::DeviceNode *devnode = (uORB::DeviceNode *)handle;
329 
330  /*
331  * We are cheating a bit here. First, with the current implementation, we can only
332  * have multiple publishers for instance 0. In this case the caller will have
333  * instance=nullptr and _published has no effect at all. Thus no unadvertise is
334  * necessary.
335  * In case of multiple instances, we have at most 1 publisher per instance and
336  * we can signal an instance as 'free' by setting _published to false.
337  * We never really free the DeviceNode, for this we would need reference counting
338  * of subscribers and publishers. But we also do not have a leak since future
339  * publishers reuse the same DeviceNode object.
340  */
341  devnode->_advertised = false;
342 
343  return 0;
344 }
