46 bool Subscription::subscribe()
47 {
48  // check if already subscribed
49  if (_node != nullptr) {
50  return true;
51  }
52 
53  if (_orb_id != ORB_ID::INVALID) {
54 
55  DeviceMaster *device_master = uORB::Manager::get_instance()->get_device_master();
56 
57  if (device_master != nullptr) {
58 
59  if (!device_master->deviceNodeExists(_orb_id, _instance)) {
60  return false;
61  }
62 
63  uORB::DeviceNode *node = device_master->getDeviceNode(get_topic(), _instance);
64 
65  if (node != nullptr) {
66  _node = node;
67  _node->add_internal_subscriber();
68 
69  const unsigned curr_gen = _node->published_message_count();
70 
71  // If there were any previous publications allow the subscriber to read them
73  _last_generation = curr_gen - !(curr_gen<(unsigned)_node->get_queue_size())?(unsigned)_node->get_queue_size():curr_gen;
74 
75  return true;
76  }
77  }
78  }
79 
80  return false;
81 }
