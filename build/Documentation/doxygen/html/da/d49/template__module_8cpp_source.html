91 {
92  int example_param = 0;
93  bool example_flag = false;
94  bool error_flag = false;
95 
96  int option_index = 1;
97  int ch;
98 
99  static struct option long_options[] = {
100  {"example_arg_1", required_argument, 0, 0 },
101  {"example_arg_2", no_argument, 0, 0 },
102  {"example_arg_3", required_argument, 0, 0 },
103  {0, 0, 0, 0 }
104  };
105 
106  // parse CLI arguments
107  while ((ch = getopt_long(argc, argv, "p:f", long_options, &option_index)) != EOF) {
108  switch (ch) {
109  case 'p':
110  example_param = 10;
111  break;
112 
113  case 'f':
114  example_flag = true;
115  break;
116 
117  case '?':
118  error_flag = true;
119  break;
120 
121  default:
122  printf("WARN: unrecognized flag");
123  error_flag = true;
124  break;
125  }
126  }
127 
128  if (error_flag) {
129  return nullptr;
130  }
131 
132  TemplateModule *instance = new TemplateModule(example_param, example_flag);
133 
134  if (instance == nullptr) {
135  printf("ERROR: alloc failed");
136  }
137 
138  return instance;
139 }
