63  void add(T newNode)
64  {
65  if (_head == nullptr) {
66  // list is empty, add as head
67  _head = newNode;
68  return;
69 
70  } else {

72  newNode->setSortedSibling(_head);
73  _head = newNode;
74  return;

87  }

98  {
